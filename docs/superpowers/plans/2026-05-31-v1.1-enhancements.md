```python
"""分享服务 —— 分析结果持久化与短链接生成。"""

import json
import secrets
import sqlite3
from pathlib import Path

DB_PATH = Path(__file__).resolve().parent.parent / "data" / "shares.db"


def _ensure_db():
    """确保数据库文件和表存在。"""
    DB_PATH.parent.mkdir(parents=True, exist_ok=True)
    conn = sqlite3.connect(str(DB_PATH))
    conn.execute("""
        CREATE TABLE IF NOT EXISTS shares (
            id             TEXT PRIMARY KEY,
            girl_name      TEXT,
            user_name      TEXT,
            overall_score  INTEGER,
            dimensions     TEXT,
            summary        TEXT,
            suggestion     TEXT,
            search_results TEXT,
            created_at     TEXT DEFAULT (datetime('now'))
        )
    """)
    conn.commit()
    return conn


def _generate_id() -> str:
    """生成 6 位短码。"""
    while True:
        raw = secrets.token_urlsafe(4)
        short = raw.replace("-", "").replace("_", "")[:6]
        if len(short) == 6:
            return short


def save(data: dict) -> str:
    """保存分享数据，返回短 ID。"""
    conn = _ensure_db()
    share_id = _generate_id()
    conn.execute(
        """INSERT INTO shares
           (id, girl_name, user_name, overall_score, dimensions, summary, suggestion, search_results)
           VALUES (?, ?, ?, ?, ?, ?, ?, ?)""",
        (
            share_id,
            data.get("girl_name", ""),
            data.get("user_name", ""),
            data.get("overall_score", 0),
            json.dumps(data.get("dimensions", []), ensure_ascii=False),
            data.get("summary", ""),
            data.get("suggestion", ""),
            json.dumps(data.get("search_results", []), ensure_ascii=False),
        ),
    )
    conn.commit()
    conn.close()
    return share_id


def get(share_id: str) -> dict | None:
    """根据短 ID 获取分享数据，不存在返回 None。"""
    conn = _ensure_db()
    row = conn.execute(
        "SELECT * FROM shares WHERE id = ?", (share_id,)
    ).fetchone()
    conn.close()
    if row is None:
        return None
    return {
        "id": row[0],
        "girl_name": row[1],
        "user_name": row[2],
        "overall_score": row[3],
        "dimensions": json.loads(row[4]),
        "summary": row[5],
        "suggestion": row[6],
        "search_results": json.loads(row[7]) if row[7] else [],
    }
```

@app.post("/api/share")
async def create_share(request: Request):
    """接收 JSON 分析结果，保存并返回短 ID。"""
    import json as _json
    body = await request.body()
    data = _json.loads(body)
    share_id = save_share(data)

@app.get("/s/{share_id}", response_class=HTMLResponse)
async def view_share(request: Request, share_id: str):

    data = get_share(share_id)

    return templates.TemplateResponse("share.html", {
        "request": request,
        **data,

替换 `form.addEventListener('submit', ...)` 部分：

```javascript
        // 异步搜索 → 展示结果 → 确认 → 提交分析
        const searchOverlay = document.getElementById('search-overlay');
        const searchOverlayText = document.getElementById('search-overlay-text');
        const searchCount = document.getElementById('search-count');
        const searchResultsWrap = document.getElementById('search-results-wrap');
        const searchResultCount = document.getElementById('search-result-count');
        const searchResultsList = document.getElementById('search-results-list');
        const searchSkip = document.getElementById('search-skip');
        const searchConfirm = document.getElementById('search-confirm');

        form.addEventListener('submit', function (e) {
            e.preventDefault();
            var enableSearch = document.getElementById('enable_search').checked;
            if (!enableSearch) {
                submitAnalysis('[]');
                return;
            }
            // 显示搜索动画
            searchOverlay.classList.add('visible');
            searchOverlayText.textContent = '正在搜索她的公开信息…';
            searchCount.textContent = '';
            var bar = searchOverlay.querySelector('.search-progress-bar');
            bar.style.animation = 'none'; bar.offsetHeight;
            bar.style.animation = 'search-progress-anim 1.8s ease-out forwards';
            submitBtn.disabled = true;
            submitBtn.textContent = '搜索中…';

            var fd = new FormData(form);
            fetch('/api/search', { method: 'POST', body: fd })
                .then(function (r) { return r.json(); })
                .then(function (results) {
                    searchOverlay.classList.remove('visible');
                    if (!results || !results.length) {
                        submitAnalysis('[]'); return;
                    }
                    searchResultCount.textContent = '找到 ' + results.length + ' 条搜索结果，勾选要参考的内容：';
                    searchResultsList.innerHTML = '';
                    results.forEach(function (r, i) {
                        var card = document.createElement('div');
                        card.className = 'search-result-card checked';
                        card.innerHTML =
                            '<input type="checkbox" checked data-idx="' + i + '">' +
                            '<div class="search-result-body"><strong>' + r.title + '</strong><p>' + (r.snippet || '') + '</p></div>';
                        card.addEventListener('click', function (ev) {
                            if (ev.target.tagName === 'INPUT') return;
                            var cb = card.querySelector('input');
                            cb.checked = !cb.checked;
                            card.classList.toggle('checked', cb.checked);
                        });
                        searchResultsList.appendChild(card);
                    });
                    searchResultsWrap.classList.add('visible');
                    submitBtn.style.display = 'none';
                })
                .catch(function () {
                    searchOverlay.classList.remove('visible');
                    submitAnalysis('[]');
                });
        });

        function submitAnalysis(selectedJson) {
            searchResultsWrap.classList.remove('visible');
            submitBtn.style.display = 'block';
            submitBtn.disabled = true;
            submitBtn.textContent = 'AI 正在分析中…';
            searchOverlay.classList.add('visible');
            searchOverlayText.textContent = '正在解读两颗心的距离…';
            searchCount.textContent = '';
            var bar = searchOverlay.querySelector('.search-progress-bar');
            bar.style.animation = 'none'; bar.offsetHeight;
            bar.style.animation = 'search-progress-anim 1.8s ease-out forwards';

            var fd = new FormData(form);
            fd.append('selected_results', selectedJson);
            fetch('/api/analyze', { method: 'POST', body: fd })
                .then(function (r) { return r.text(); })
                .then(function (html) {
                    document.open(); document.write(html); document.close();
                });
        }

        searchSkip.addEventListener('click', function () {
            submitAnalysis('[]');
        });
        searchConfirm.addEventListener('click', function () {
            var checked = [];
            document.querySelectorAll('.search-result-card input:checked').forEach(function (cb) {
                var idx = parseInt(cb.getAttribute('data-idx'));
                var titleEl = cb.parentElement.querySelector('strong');
                var snippetEl = cb.parentElement.querySelector('p');
                checked.push({ title: titleEl.textContent, url: '', snippet: snippetEl ? snippetEl.textContent : '' });
            });
            submitAnalysis(JSON.stringify(checked));
        });
```
